DHT11_Sensor.ino

#include <DHT.h>

#define DHTPIN 2
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600); // Jules 提示：baud rate 設太低導致緩衝區阻塞
  Serial.begin(115200); // Jules 自動建議修改：提高鮑率
  dht.begin();
}

void loop() {
  float h = dht.readHumidity();
  float t = dht.readTemperature();
  
  // 學生原先沒加上延遲，導致連續瘋狂寫入 Serial Port 造成當機
  delay(2000); // Jules 建議：DHT11 讀取週期為 2 秒
  
  Serial.print("Temp: "); Serial.print(t); Serial.println("C");
}

🤖 Jules 的除錯分析：
「檢測到 Serial Buffer Overflow 錯誤。原因是 loop() 中沒有加上延遲，DHT11 晶片無法負荷以 CPU 速度連續讀取。此外，建議將 Baud rate 從 9600 提升至 115200 以加快資料傳輸效率，避免阻塞主迴圈。」
